(SELECT   MAX(NULLIF ('X', '')) OFERTA
             FROM   PCOFERTAPROGRAMADAC OC, PCOFERTAPROGRAMADAI OI
            WHERE       OC.CODOFERTA = OI.CODOFERTA
                    AND OC.CODFILIAL = OI.CODFILIAL
                    AND MV.DTMOV BETWEEN OC.DTINICIAL AND OC.DTFINAL
                    AND MV.CODPROD = OI.CODPROD
                    AND MV.CODAUXILIAR = OI.CODAUXILIAR
                    AND (MV.PUNIT = OI.VLOFERTAATAC OR MV.PUNIT = OI.VLOFERTA)
                    AND MV.CODFILIALNF = OI.CODFILIAL)
              OFERTA
